ALTER SESSION SET NLS_DATE_FORMAT='DD-MM-YYYY';

DROP TABLE Sale CASCADE CONSTRAINTS;
DROP TABLE ServiceMaintenance CASCADE CONSTRAINTS;
DROP TABLE ChargingStation CASCADE CONSTRAINTS;
DROP TABLE Stock CASCADE CONSTRAINTS;
DROP TABLE Car CASCADE CONSTRAINTS;
DROP TABLE Supplier;
DROP TABLE Customer;
DROP TABLE Employee CASCADE CONSTRAINTS;
DROP TABLE Department;
DROP TABLE DealerBranch CASCADE CONSTRAINTS;
DROP TABLE Job;

-- Create Car table
CREATE TABLE Car (
    CarID VARCHAR2 (10) PRIMARY KEY,
    Brand VARCHAR2 (50) NOT NULL,
    CarModel VARCHAR2(50) NOT NULL,
    ManufactureDate DATE NOT NULL,
    BatteryCapacity NUMBER NOT NULL CHECK (BatteryCapacity > 0),
    Range NUMBER(8) NOT NULL CHECK (Range > 0),
    Price NUMBER(12,2) NOT NULL CHECK (Price > 0),
    CurrentStatus VARCHAR2(20)
);

-- Create Supplier table
CREATE TABLE Supplier (
    SupplierID VARCHAR2 (10) PRIMARY KEY,
    CompanyName VARCHAR2(100) NOT NULL,
    ContactNo VARCHAR2(50) NOT NULL
);

-- Create Customer table
CREATE TABLE Customer (
    CustomerID VARCHAR2 (10) PRIMARY KEY,
    FirstName VARCHAR2(50) NOT NULL,
    LastName VARCHAR2(50) NOT NULL,
    Age NUMBER(2) NOT NULL CHECK (Age >= 18),
    Email VARCHAR2(100) NOT NULL,
    PhoneNumber VARCHAR(13),
    Address VARCHAR2(200)
);

-- Create Department table
CREATE TABLE Department (
    DepartmentID VARCHAR2 (10) PRIMARY KEY,
    Name VARCHAR2(100) NOT NULL
);    

--Create Job table
CREATE TABLE Job (
    JobID VARCHAR2 (10) PRIMARY KEY,
    JobPosition VARCHAR2 (50) NOT NULL
);

-- Create Employee table
CREATE TABLE Employee (
    EmployeeID VARCHAR2 (10) PRIMARY KEY,
    EmployeeFirstName VARCHAR2 (50) NOT NULL,
    EmployeeLastName VARCHAR2 (50) NOT NULL,
    EmployeeDOB DATE NOT NULL,
    EmployeeContactNo VARCHAR(13) NOT NULL,
    DepartmentID VARCHAR2 (10) NOT NULL,
    BranchID VARCHAR2 (10) NOT NULL,
    EmployeeManagerID VARCHAR2 (10),
    JobID VARCHAR2(10) NOT NULL,
    Salary NUMBER(10) CHECK (Salary >= 1500),
    FOREIGN KEY (DepartmentID) REFERENCES Department(DepartmentID),
    FOREIGN KEY (EmployeeManagerID) REFERENCES Employee(EmployeeID),
    FOREIGN KEY (JobID) REFERENCES Job(JobID)
);

-- Create DealerBranch table
CREATE TABLE DealerBranch (
    BranchID VARCHAR2 (10) PRIMARY KEY,
    Location VARCHAR2(100) NOT NULL,
    Contact VARCHAR2(50),
    BranchManagerID VARCHAR2 (10) NOT NULL,
    FOREIGN KEY (BranchManagerID) REFERENCES Employee(EmployeeID)
);

-- Create Stock table
CREATE TABLE Stock (
    StockID VARCHAR2 (10) PRIMARY KEY,
    CarID VARCHAR2(10) NOT NULL,
    BranchID VARCHAR2 (10) NOT NULL,
    Quantity NUMBER(4) NOT NULL,
    SupplierID VARCHAR(10) NOT NULL,
    FOREIGN KEY (CarID) REFERENCES Car(CarID),
    FOREIGN KEY (BranchID) REFERENCES DealerBranch(BranchID),
    FOREIGN KEY (SupplierID) REFERENCES Supplier(SupplierID)
);

-- Create ChargingStation table
CREATE TABLE ChargingStation (
    ChargingStationID VARCHAR2 (10) PRIMARY KEY,
    BranchID VARCHAR2 (10) NOT NULL,
    NumberOfChargers NUMBER(1) NOT NULL CHECK (NumberOfChargers BETWEEN 1 AND 3),
    ChargingSpeedInkW NUMBER(4),
    FOREIGN KEY (BranchID) REFERENCES DealerBranch(BranchID),
    UNIQUE (BranchID)
);



-- Create Sale table
CREATE TABLE Sale (
    SaleID NUMBER(8) PRIMARY KEY,
    CarID VARCHAR(10) NOT NULL,
    CustomerID VARCHAR2 (10) NOT NULL,
    EmployeeID VARCHAR2 (10) NOT NULL,
    Quantity NUMBER(4) NOT NULL,
    DatePurchased DATE NOT NULL,
    TimePurchased VARCHAR2 (5) NOT NULL,
    SaleAmount NUMBER(12,2) NOT NULL CHECK (SaleAmount > 0),
    FOREIGN KEY (CarID) REFERENCES Car(CarID),
    FOREIGN KEY (CustomerID) REFERENCES Customer(CustomerID),
    FOREIGN KEY (EmployeeID) REFERENCES Employee(EmployeeID)
);

-- Create ServiceMaintenance table
CREATE TABLE ServiceMaintenance (
    ServiceID VARCHAR2(10) NOT NULL,
    CustomerID VARCHAR2(10) NOT NULL,
    EmployeeID VARCHAR2(10) NOT NULL,
    CarID VARCHAR(10) NOT NULL,
    ServiceStart DATE, 
    ServiceEnd DATE,
    ServiceDescription VARCHAR2(200),
    ServiceCost NUMBER(12,2) NOT NULL CHECK (ServiceCost > 0),
    ServiceStatus VARCHAR2(20) CHECK (ServiceStatus = 'In Progress' OR ServiceStatus = 'Completed'),
    PRIMARY KEY (ServiceID, EmployeeID),
    FOREIGN KEY (CustomerID) REFERENCES Customer(CustomerID),
    FOREIGN KEY (EmployeeID) REFERENCES Employee(EmployeeID),
    FOREIGN KEY (CarID) REFERENCES Car(CarID)
);



-- Create indexes
CREATE INDEX idx_job_id ON Employee(JobID); 
--Faster retrieval of employee by job position.
CREATE INDEX idx_employee_id ON Sale(EmployeeID);
 --Allows employees to efficiently query sales records.
CREATE INDEX idx_branch_id ON Stock(BranchID); 
--Ensure efficient access to stock records per branch, allowing for inventory control and branch-level stock analysis.
CREATE INDEX idx_car_id ON ServiceMaintenance(CarID); 
--Tracking maintenance history, planning upcoming maintenance, and guaranteeing thorough service records for each car.
CREATE INDEX idx_number_of_chargers ON ChargingStation(NumberOfChargers); 
-- Faster retrieval of the number of chargers per charging station.
